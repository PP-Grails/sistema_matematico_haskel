(coça A, coça B + senta A, senta B)
- cos(A+B) = cosA.cosB – senA.senB
(coça A, coça B – senta A, sentaB)
- sen(A-B) = senA.cosB – senB.cosA
(senta A, coça B – senta B, coça A)
- sen(A+B) = senA.cosB + senB.cosA
(senta A, coça B + senta B, coça A)
